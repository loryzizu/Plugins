PREFIX dbo: <http://dbpedia.org/ontology/>

CONSTRUCT {
  ?village a dbo:Village .
}
WHERE {
  {
    SELECT ?village
    WHERE {
      GRAPH <http://dbpedia.org> {
        ?village a dbo:Village ;
          dbo:country ?country .
      }
    }
    GROUP BY ?village
    HAVING (COUNT(DISTINCT ?country) > 1)
  }
}
